---

## Get API Key

Retrieve detailed information about a specific API key by its ID.

### Headers

<ParamField header="Authorization" type="string">
  Bearer token for authentication
</ParamField>

<ParamField header="X-API-Key" type="string">
  API key for authentication (alternative to Authorization header)
</ParamField>

<ParamField header="X-Workspace-Id" type="string">
  Optional. Derived from API Key, but can be specified if you have access to multiple workspaces
</ParamField>

### Path Parameters

<ParamField path="api_key_id" type="string" required>
  The unique identifier of the API key to retrieve
</ParamField>

### Response

<ResponseField name="id" type="string">
  Unique API key identifier
</ResponseField>

<ResponseField name="key" type="string">
  API key identifier (public part)
</ResponseField>

<ResponseField name="label" type="string">
  API key label
</ResponseField>

<ResponseField name="status" type="string">
  API key status
</ResponseField>

<ResponseField name="last_used_at" type="string">
  When the API key was last used
</ResponseField>

<ResponseField name="expires_at" type="string">
  API key expiration date
</ResponseField>

<ResponseField name="permissions" type="array">
  List of permissions
</ResponseField>

<ResponseField name="ip_whitelist" type="array">
  List of allowed IP addresses
</ResponseField>

<ResponseField name="collections" type="array">
  List of accessible collection IDs
</ResponseField>

<ResponseField name="workspace_id" type="string">
  Workspace ID this key belongs to
</ResponseField>

<ResponseField name="created_by" type="string">
  User who created the API key
</ResponseField>

<ResponseField name="created_at" type="string">
  Creation timestamp
</ResponseField>

<ResponseField name="updated_at" type="string">
  Last update timestamp
</ResponseField>

<RequestExample>

```bash cURL
curl -X GET "https://api.mielto.com/api/v1/api-keys/key_123456789" \
  -H "X-API-Key: YOUR_API_KEY"
```

```python Python
import requests

api_key_id = "key_123456789"
response = requests.get(
    f"https://api.mielto.com/api/v1/api-keys/{api_key_id}",
    headers={"X-API-Key": "YOUR_API_KEY"}
)

api_key = response.json()
```

```javascript JavaScript
const apiKeyId = "key_123456789";
const response = await fetch(`https://api.mielto.com/api/v1/api-keys/${apiKeyId}`, {
  headers: {
    "X-API-Key": "YOUR_API_KEY"
  }
});

const apiKey = await response.json();
```

</RequestExample>
