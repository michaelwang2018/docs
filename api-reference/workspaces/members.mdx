## Get Workspace Members

Retrieve detailed workspace information including all member users, their roles, and permissions.

### Headers

<ParamField header="X-API-Key" type="string" required>
  Your API key for authentication
</ParamField>

<ParamField header="X-Workspace-Id" type="string">
  Optional workspace ID header
</ParamField>

### Path Parameters

<ParamField path="workspace_id" type="string" required>
  The unique identifier of the workspace
</ParamField>

### Response

<ResponseField name="id" type="string">
  Unique workspace identifier
</ResponseField>

<ResponseField name="name" type="string">
  Workspace name
</ResponseField>

<ResponseField name="description" type="string">
  Workspace description
</ResponseField>

<ResponseField name="status" type="string">
  Workspace status
</ResponseField>

<ResponseField name="members" type="array">
  Array of workspace members
  
  <Expandable title="member object">
    <ResponseField name="id" type="string">
      Workspace user ID
    </ResponseField>
    
    <ResponseField name="user_id" type="string">
      Reference to user
    </ResponseField>
    
    <ResponseField name="role" type="string">
      User role in workspace (owner, admin, member, viewer, developer)
    </ResponseField>
    
    <ResponseField name="permissions" type="object">
      User permissions
    </ResponseField>
    
    <ResponseField name="status" type="string">
      User status in workspace
    </ResponseField>
    
    <ResponseField name="is_owner" type="boolean">
      Whether user is workspace owner
    </ResponseField>
    
    <ResponseField name="invited_by" type="string">
      User who sent the invitation
    </ResponseField>
    
    <ResponseField name="invited_at" type="string">
      When the invitation was sent
    </ResponseField>
  </Expandable>
</ResponseField>

<RequestExample>

```bash cURL
curl -X GET "https://api.mielto.com/api/v1/workspaces/wsp_123456789/members" \
  -H "X-API-Key: YOUR_API_KEY"
```

```python Python
import requests

workspace_id = "wsp_123456789"
response = requests.get(
    f"https://api.mielto.com/api/v1/workspaces/{workspace_id}/members",
    headers={"X-API-Key": "YOUR_API_KEY"}
)

workspace_with_members = response.json()
```

```javascript JavaScript
const workspaceId = "wsp_123456789";
const response = await fetch(`https://api.mielto.com/api/v1/workspaces/${workspaceId}/members`, {
  headers: {
    "X-API-Key": "YOUR_API_KEY"
  }
});

const workspaceWithMembers = await response.json();
```
