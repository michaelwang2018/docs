## Get Collection

Retrieve detailed information about a specific collection by its ID.

### Headers

<ParamField header="Authorization" type="string">
  Bearer token for authentication
</ParamField>

<ParamField header="X-API-Key" type="string">
  API key for authentication (alternative to Authorization header)
</ParamField>

<ParamField header="X-Workspace-Id" type="string">
  Optional. Derived from API Key, but can be specified if you have access to multiple workspaces
</ParamField>

### Path Parameters

<ParamField path="collection_id" type="string" required>
  The unique identifier of the collection to retrieve
</ParamField>

### Response

<ResponseField name="id" type="string">
  Unique collection identifier
</ResponseField>

<ResponseField name="name" type="string">
  Collection name
</ResponseField>

<ResponseField name="description" type="string">
  Collection description
</ResponseField>

<ResponseField name="store_type" type="string">
  Storage type for the collection
</ResponseField>

<ResponseField name="visibility" type="string">
  Collection visibility (private, public)
</ResponseField>

<ResponseField name="status" type="string">
  Collection status (active, archived)
</ResponseField>

<ResponseField name="tags" type="array">
  Collection tags for organization
</ResponseField>

<ResponseField name="parent_id" type="string">
  Reference to parent collection (if any)
</ResponseField>

<ResponseField name="stats" type="object">
  Collection statistics
</ResponseField>

<ResponseField name="meta_data" type="object">
  Collection metadata
</ResponseField>

<ResponseField name="settings" type="object">
  Collection configuration settings
</ResponseField>

<ResponseField name="embedding" type="object">
  Embedding configuration
</ResponseField>

<ResponseField name="source_type" type="string">
  Collection source type
</ResponseField>

<ResponseField name="source_id" type="string">
  Collection source ID
</ResponseField>

<ResponseField name="workspace_id" type="string">
  Workspace ID this collection belongs to
</ResponseField>

<ResponseField name="created_by" type="string">
  User who created the collection
</ResponseField>

<ResponseField name="created_at" type="string">
  Creation timestamp
</ResponseField>

<ResponseField name="updated_at" type="string">
  Last update timestamp
</ResponseField>

<RequestExample>

```bash cURL
curl -X GET "https://api.mielto.com/api/v1/collections/col_123456789" \
  -H "X-API-Key: YOUR_API_KEY"
```

```python Python
import requests

collection_id = "col_123456789"
response = requests.get(
    f"https://api.mielto.com/api/v1/collections/{collection_id}",
    headers={"X-API-Key": "YOUR_API_KEY"}
)

collection = response.json()
```

```javascript JavaScript
const collectionId = "col_123456789";
const response = await fetch(`https://api.mielto.com/api/v1/collections/${collectionId}`, {
  headers: {
    "X-API-Key": "YOUR_API_KEY"
  }
});

const collection = await response.json();
```
