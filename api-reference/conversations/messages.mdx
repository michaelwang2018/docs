## Get Conversation Messages

Retrieve all messages within a specific conversation with pagination, filtering, and ordering options.

### Headers

<ParamField header="Authorization" type="string">
  Bearer token for authentication
</ParamField>

<ParamField header="X-API-Key" type="string">
  API key for authentication (alternative to Authorization header)
</ParamField>

<ParamField header="X-Workspace-Id" type="string">
  Optional. Derived from API Key, but can be specified if you have access to multiple workspaces
</ParamField>

### Path Parameters

<ParamField path="conversation_id" type="string" required>
  The unique identifier of the conversation to retrieve messages from
</ParamField>

### Query Parameters

<ParamField query="skip" type="integer" default="0">
  Number of messages to skip for pagination
</ParamField>

<ParamField query="limit" type="integer" default="100">
  Maximum number of messages to return (1-1000)
</ParamField>

<ParamField query="role" type="string">
  Filter messages by role (user, assistant, system)
</ParamField>

<ParamField query="message_type" type="string">
  Filter messages by type
</ParamField>

<ParamField query="order" type="string" default="asc">
  Order messages by timestamp (asc or desc)
</ParamField>

### Response

<ResponseField name="data" type="array">
  Array of message objects
  
  <Expandable title="message object">
    <ResponseField name="id" type="string">
      Unique message identifier
    </ResponseField>
    
    <ResponseField name="conversation_id" type="string">
      ID of the conversation this message belongs to
    </ResponseField>
    
    <ResponseField name="content" type="string">
      Message content text
    </ResponseField>
    
    <ResponseField name="role" type="string">
      Message role (user, assistant, system)
    </ResponseField>
    
    <ResponseField name="message_type" type="string">
      Type of message
    </ResponseField>
    
    <ResponseField name="sender_id" type="string">
      ID of the message sender
    </ResponseField>
    
    <ResponseField name="meta_data" type="object">
      Additional message metadata
    </ResponseField>
    
    <ResponseField name="tokens" type="integer">
      Token count for this message
    </ResponseField>
    
    <ResponseField name="created_at" type="string">
      Timestamp when message was created
    </ResponseField>
    
    <ResponseField name="updated_at" type="string">
      Timestamp when message was last updated
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="total_count" type="integer">
  Total number of messages in the conversation
</ResponseField>

<ResponseField name="conversation" type="object">
  Basic conversation information
  
  <Expandable title="conversation info">
    <ResponseField name="id" type="string">
      Conversation ID
    </ResponseField>
    
    <ResponseField name="title" type="string">
      Conversation title
    </ResponseField>
    
    <ResponseField name="status" type="string">
      Conversation status
    </ResponseField>
  </Expandable>
</ResponseField>

<RequestExample>

```bash cURL
curl "https://api.mielto.com/api/v1/conversations/conv_123456789/messages?skip=0&limit=20&order=asc" \
  -H "X-API-Key: YOUR_API_KEY"
```

```python Python
import requests

conversation_id = "conv_123456789"
response = requests.get(
    f"https://api.mielto.com/api/v1/conversations/{conversation_id}/messages",
    headers={"X-API-Key": "YOUR_API_KEY"},
    params={
        "skip": 0,
        "limit": 20,
        "order": "asc"
    }
)

messages = response.json()
```

```javascript JavaScript
const conversationId = "conv_123456789";
const response = await fetch(`https://api.mielto.com/api/v1/conversations/${conversationId}/messages?skip=0&limit=20&order=asc`, {
  headers: {
    "X-API-Key": "YOUR_API_KEY"
  }
});

const messages = await response.json();
```
