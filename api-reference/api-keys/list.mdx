## List API Keys

Retrieve a paginated list of all API keys in your workspace.

### Headers

<ParamField header="Authorization" type="string">
  Bearer token for authentication
</ParamField>

<ParamField header="X-API-Key" type="string">
  API key for authentication (alternative to Authorization header)
</ParamField>

<ParamField header="X-Workspace-Id" type="string">
  Optional. Derived from API Key, but can be specified if you have access to multiple workspaces
</ParamField>

### Query Parameters

<ParamField query="skip" type="integer" default="0">
  Number of API keys to skip for pagination (minimum: 0)
</ParamField>

<ParamField query="limit" type="integer" default="100">
  Maximum number of API keys to return (1-1000)
</ParamField>

### Response

<ResponseField name="data" type="array">
  Array of API key objects
  
  <Expandable title="API key object">
    <ResponseField name="id" type="string">
      Unique API key identifier
    </ResponseField>
    
    <ResponseField name="key" type="string">
      API key identifier (public part)
    </ResponseField>
    
    <ResponseField name="label" type="string">
      API key label
    </ResponseField>
    
    <ResponseField name="status" type="string">
      API key status
    </ResponseField>
    
    <ResponseField name="last_used_at" type="string">
      When the API key was last used
    </ResponseField>
    
    <ResponseField name="expires_at" type="string">
      API key expiration date
    </ResponseField>
    
    <ResponseField name="permissions" type="array">
      List of permissions
    </ResponseField>
    
    <ResponseField name="collections" type="array">
      List of accessible collection IDs
    </ResponseField>
    
    <ResponseField name="created_at" type="string">
      Creation timestamp
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="total_count" type="integer">
  Total number of API keys
</ResponseField>

<RequestExample>

```bash cURL
curl -X GET "https://api.mielto.com/api/v1/api-keys?skip=0&limit=50" \
  -H "X-API-Key: YOUR_API_KEY"
```

```python Python
import requests

response = requests.get(
    "https://api.mielto.com/api/v1/api-keys",
    headers={"X-API-Key": "YOUR_API_KEY"},
    params={"skip": 0, "limit": 50}
)

api_keys = response.json()
```

```javascript JavaScript
const response = await fetch("https://api.mielto.com/api/v1/api-keys?skip=0&limit=50", {
  headers: {
    "X-API-Key": "YOUR_API_KEY"
  }
});

const apiKeys = await response.json();
```

</RequestExample>
