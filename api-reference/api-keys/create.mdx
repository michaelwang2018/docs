## Create API Key

Create a new API key for programmatic access to your workspace. The API key will be used to authenticate API requests.

### Headers

<ParamField header="Authorization" type="string">
  Bearer token for authentication
</ParamField>

<ParamField header="X-API-Key" type="string">
  API key for authentication (alternative to Authorization header)
</ParamField>

<ParamField header="X-Workspace-Id" type="string">
  Optional. Derived from API Key, but can be specified if you have access to multiple workspaces
</ParamField>

### Request Body

<ParamField body="label" type="string" required>
  Human-readable label for the API key (1-255 characters)
</ParamField>

<ParamField body="status" type="string" default="active">
  API key status (active, inactive)
</ParamField>

<ParamField body="expires_at" type="string">
  API key expiration date (ISO 8601 format)
</ParamField>

<ParamField body="permissions" type="array">
  List of permissions granted to this API key
</ParamField>

<ParamField body="ip_whitelist" type="array">
  List of allowed IP addresses for this API key
</ParamField>

<ParamField body="collections" type="array">
  List of collection IDs this API key can access
</ParamField>

### Response

<ResponseField name="id" type="string">
  Unique API key identifier
</ResponseField>

<ResponseField name="key" type="string">
  API key identifier (public part)
</ResponseField>

<ResponseField name="secret" type="string">
  API key secret (only returned on creation)
</ResponseField>

<ResponseField name="label" type="string">
  API key label
</ResponseField>

<ResponseField name="status" type="string">
  API key status
</ResponseField>

<ResponseField name="permissions" type="array">
  List of permissions
</ResponseField>

<ResponseField name="expires_at" type="string">
  Expiration date
</ResponseField>

<ResponseField name="workspace_id" type="string">
  Workspace ID this key belongs to
</ResponseField>

<ResponseField name="created_at" type="string">
  Creation timestamp
</ResponseField>

<RequestExample>

```bash cURL
curl -X POST "https://api.mielto.com/api/v1/api-keys" \
  -H "X-API-Key: YOUR_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{
    "label": "Production API Key",
    "status": "active",
    "expires_at": "2025-12-31T23:59:59Z",
    "permissions": ["read", "write"],
    "collections": ["col_123", "col_456"]
  }'
```

```python Python
import requests

response = requests.post(
    "https://api.mielto.com/api/v1/api-keys",
    headers={
        "X-API-Key": "YOUR_API_KEY",
        "Content-Type": "application/json"
    },
    json={
        "label": "Production API Key",
        "status": "active",
        "expires_at": "2025-12-31T23:59:59Z",
        "permissions": ["read", "write"],
        "collections": ["col_123", "col_456"]
    }
)

api_key = response.json()
print(f"New API Key: {api_key['secret']}")
```

```javascript JavaScript
const response = await fetch("https://api.mielto.com/api/v1/api-keys", {
  method: "POST",
  headers: {
    "X-API-Key": "YOUR_API_KEY",
    "Content-Type": "application/json"
  },
  body: JSON.stringify({
    label: "Production API Key",
    status: "active",
    expires_at: "2025-12-31T23:59:59Z",
    permissions: ["read", "write"],
    collections: ["col_123", "col_456"]
  })
});

const apiKey = await response.json();
console.log(`New API Key: ${apiKey.secret}`);
```
